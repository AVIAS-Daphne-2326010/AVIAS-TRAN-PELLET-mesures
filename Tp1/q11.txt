// Fonction de quicksort

void iqsort0 ( int *a , int n )
{
	int i , j ;
	// condition d'arret
	if ( n <=1)
		return ;
	// boucle de permutation de valeur
	for ( i = 1 , j = 0; i < n ; i ++)
		if ( a [ i ] < a [0])
			swap (++ j , i , a );
	swap (0 , j , a );
	// trie recursif de la partie gauche
	iqsort0 (a , j );		
	//trie recursif de la partie droite
	iqsort0 ( a + j +1 , n -j -1;
}
